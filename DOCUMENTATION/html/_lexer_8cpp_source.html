222 cursor_two = 0;

226 while (cursor_two <= line_len) {
227 cur_char = line[cursor_two];
228 if (state == S_first) {
229 // First check for identifier or number
230 if (std::isalpha(cur_char) || cur_char == '_') state = S_identifier;
231 else if (std::isdigit(cur_char)) state = S_number;
232 else state = S_other;

236 distance = cursor_two - cursor_one;
